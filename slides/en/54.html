Read and write operations have been moved

  • The traits to represent read and write operations are now in the futures crate
pub trait AsyncWrite {
    fn poll_write(
        &mut self,

    fn poll_flush(&mut self, cx: &mut Context) -> Result<Async<()>, Error>;

    fn poll_close(&mut self, cx: &mut Context) -> Result<Async<()>, Error>;

    fn poll_vectored_write(
        &mut self, 
        cx: &mut Context, 
        vec: &[&IoVec]
    ) -> Result<Async<usize>, Error> { ... }
}
