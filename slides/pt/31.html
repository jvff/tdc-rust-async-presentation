Canal mpsc

  • mpsc permite enviar vários ítens
  • Várias tarefas podem enviar mas somente uma pode receber
  • mpsc::Sender implementa Sink
  • mpsc::Receiver implementa Stream

let (sender, receiver) = mpsc::unbounded();

let sender2 = sender.clone();

sender.send(item1).wait();
sender2.send(item2).wait();

for item in receiver.wait() {
    // ...
}
