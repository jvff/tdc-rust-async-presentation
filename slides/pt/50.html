Contexto evidente

  • Agora fica mais claro que uma operação assíncrona está diretamente associada ao contexto de uma tarefa
pub trait Future {
    type Item;
    type Error;

    fn poll(
        &mut self, 
        cx: &mut Context
    ) -> Result<Async<Self::Item>, Self::Error>;
}

pub trait Stream {
    type Item;
    type Error;

    fn poll_next(
        &mut self, 
        cx: &mut Context
    ) -> Result<Async<Option<Self::Item>>, Self::Error>;
}
